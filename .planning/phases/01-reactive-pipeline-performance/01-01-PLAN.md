# Plan 01-01: Deferred Entity Subscription + Store Mutation Guards

## Objective

Stop subscribing to all entities on connect. Defer subscription until first widget registers an entity. Add equality checks to prevent SolidJS store proxy from firing on identity writes.

## must_haves

- Empty dashboard receives zero entity subscription messages after initial `get_states` load
- Entity heartbeat updates with only `last_updated` changes do not trigger store property mutations
- `applyStateDiff` does not write `existing.state` when value is already identical

   - Remove the line `const initialUnsub = await subscribeEntities(conn, []);`
   - Remove `setCurrentEntityUnsub(initialUnsub);`
   - The subscription manager will handle the first subscribe when widgets register entities via `registerEntity()`.

4. In `subscription-manager.ts` `flush()`, handle the transition from zeroâ†’nonzero:
   ```typescript
   async function flush(): Promise<void> {
     if (!conn || !onResubscribe) return;
     const newIds = new Set(refCounts.keys());
     
     // No entities needed â€” unsubscribe if active
     if (newIds.size === 0) {
       if (currentEntityUnsub) {
         const oldUnsub = currentEntityUnsub;
         currentEntityUnsub = null;
         subscribedIds = new Set();
         try { await oldUnsub(); } catch { /* dead subscription */ }
       }
       return;
     }
     
     // Skip if set unchanged
     if (setsEqual(newIds, subscribedIds)) return;
     
     // Subscribe/resubscribe
     const oldUnsub = currentEntityUnsub;
     currentEntityUnsub = await onResubscribe(conn, [...newIds]);
     subscribedIds = newIds;
     if (oldUnsub) {
       try { await oldUnsub(); } catch { /* dead subscription */ }
     }
   }
   ```
</action>

</acceptance_criteria>

<read_first>
- packages/public/sync-layer/src/connection/subscriptions.ts (lines 509-565)

<action>
1. In `applyStateDiff` (line 532-565), guard each property write:

   ```typescript
   function applyStateDiff(
     s: import("../core/store").GlassHomeState,
     entityId: string,
     diff: EntityStateDiff,
   ): void {
     const existing = s.entities[entityId];
     if (!existing) return;

     const additions = diff["+"];
     if (additions) {
       if (additions.s !== undefined && additions.s !== existing.state) {
         existing.state = additions.s;
       }
       if (additions.a) {
         for (const [key, value] of Object.entries(additions.a)) {
           if (existing.attributes[key] !== value) {
             existing.attributes[key] = value;
           }
         }
       }
       if (additions.lc !== undefined) {
         const newLc = timestampToIso(additions.lc);
         if (newLc !== existing.last_changed) {
           existing.last_changed = newLc;
         }
       }
       if (additions.lu !== undefined) {
         const newLu = timestampToIso(additions.lu);
         if (newLu !== existing.last_updated) {
           existing.last_updated = newLu;
         }
       }
       if (additions.c) {
         existing.context = additions.c;
       }
     }

     const removals = diff["-"];
     if (removals?.a) {
       for (const key of removals.a) {
         delete existing.attributes[key];
       }
     }
   }
   ```

2. In `applyCompressedState` (line 509-530), check if entity already exists with same state before replacing:

   ```typescript
   function applyCompressedState(
     s: import("../core/store").GlassHomeState,
     entityId: string,
     compressed: CompressedEntityState,
   ): void {
     const stateVal = compressed.s ?? compressed.state ?? "unknown";
     const attributes = compressed.a ?? compressed.attributes ?? {};
     const context = compressed.c ?? compressed.context ?? { id: "", parent_id: null, user_id: null };
     const lastChanged = timestampToIso(compressed.lc ?? compressed.last_changed);
     const lastUpdated = timestampToIso(compressed.lu ?? compressed.last_updated);

     const existing = s.entities[entityId];
     if (existing && existing.state === stateVal && existing.last_changed === lastChanged && existing.last_updated === lastUpdated) {
       return;
     }

     s.entities[entityId] = {
       entity_id: entityId,
       state: stateVal,
       attributes: attributes as Record<string, unknown>,
       last_changed: lastChanged,
       last_updated: lastUpdated,
       context,
     };
   }
   ```
</action>
